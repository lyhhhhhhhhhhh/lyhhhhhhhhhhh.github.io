func preorderTraversal(root *TreeNode) []int {
    stack := list.New()
    res := make([]int, 0)
    stack.PushBack(root)
    for stack.Len() > 0 {
        cur  := stack.Back()
        curValue, _ := cur.Value.(*TreeNode)
        stack.Remove(cur)
        if curValue != nil {
            res = append(res, curValue.Val)
            stack.PushBack(curValue.Right)
            stack.PushBack(curValue.Left)
        }
    }
    return res
}
