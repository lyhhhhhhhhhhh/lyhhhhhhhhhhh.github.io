func hIndex(citations []int) int {
    m := make(map[int]int)
    for _, value := range citations {
        m[value]++
    }
    store := make([]int, 0)
    for k := range m {
        store = append(store, k)
    }
    sort.Ints(store)
    sum := 0
    for i := len(store) - 1; i >= 0; i-- {
        sum += m[store[i]]
        if sum >= store[i] {
            temp := sum - m[store[i]]
            if i != len(store)-1 && temp >= store[i] {
                return temp
            }
            return store[i]
        }
    }
    if len(citations) > 0 {
        return len(citations)
    }
    return 0
}

func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    firstLength := getlength(l1)
    secondLength := getlength(l2)
    add := &ListNode{0, nil}
    tail := add
    if firstLength > secondLength {
        for i := 1; i < (firstLength - secondLength); i++ {
            tail.Next = &ListNode{0, nil}
            tail = tail.Next
        }
        tail.Next = l2
        l2 = add
    } else if firstLength < secondLength {
        for i := 1; i < (secondLength - firstLength); i++ {
            tail.Next = &ListNode{0, nil}
            tail = tail.Next
        }
        tail.Next = l1
        l1 = add
    }
    if calculate(l1, l2) == 1 {
        head := &ListNode{1, l1}
        return head
    }
    return l1
}

func getlength(root *ListNode) int {
    res := 0
    for root != nil {
        res++
        root = root.Next
    }
    return res
}

func calculate(l1, l2 *ListNode) int {
    if l1 == nil && l2 == nil {
        return 0
    }
    res := l1.Val + l2.Val + calculate(l1.Next, l2.Next)
    l1.Val = res % 10
    return res / 10
}

func myAtoi(str string) int {
    res, negative, begin := 0, false, false
    for _, value := range str {
        if value == ' ' && !begin {
            continue
        }
        if (value == '+' || value == '-') && !begin {
            if value == '-' {
                negative = true
            }
            begin = true
            continue
        }
        if value >= '0' && value <= '9' {
            if !begin {
                begin = true
            }
            res = res*10 + int(value-48)
            if res > (-math.MinInt32) && negative {
                res = math.MinInt32
                negative = false
                break
            }
            if res > math.MaxInt32 && !negative {
                res = math.MaxInt32
                break
            }
        } else {
            break
        }
    }
    if negative {
        res = (-res)
    }
    return res
}

    return MaxQueue{

        aux: make([]int, 0),
    }

func (this *MaxQueue) Max_value() int {
    if len(this.aux) == 0 {
        return -1
    }
    return this.aux[0]
}


func (this *MaxQueue) Push_back(value int)  {
    this.queue = append(this.queue, value)
    index := 0
    for index = range this.aux {
        if this.aux[index] < value {
            break
        }
    }
    if len(this.aux) > 0 && this.aux[index] >= value { index++ }
    this.aux = append(this.aux[:index], value)
}


func (this *MaxQueue) Pop_front() int {
    if len(this.queue) == 0 {
        return -1
    }
    res := this.queue[0]
    this.queue = this.queue[1:]
    if this.aux[0] == res {
        this.aux = this.aux[1:]
    }
    return res
}

func rotateRight(head *ListNode, k int) *ListNode {
    if head == nil {
        return head
    }
    travel, length := head, 1
    for travel.Next != nil {
        length++
        travel = travel.Next
    }
    travel.Next = head
    for i:=0; i<length-(k%length)-1; i++ {
        head = head.Next
    }
    head, head.Next = head.Next, nil
    return head
}

func maxAreaOfIsland(grid [][]int) int {
    max := 0
    queue:= make([]int, 0)
    for i, v := range grid {
        for j, value := range v {
            temp := 0
            if value == 1 {
                temp++
                grid[i][j] = 0
                queue = append(queue, i*100+j)
                for len(queue) > 0 {
                    nextQue := make([]int, 0)
                    for _, node := range queue {
                        cow, row := node%100, node/100
                        if row!=0 && grid[row-1][cow]==1 {
                            temp++
                            grid[row-1][cow] = 0
                            nextQue = append(nextQue, (row-1)*100+cow)
                        }
                        if row!=len(grid)-1 && grid[row+1][cow]==1 {
                            temp++
                            grid[row+1][cow]=0
                            nextQue = append(nextQue, (row+1)*100+cow)
                        }
                        if cow!=0 && grid[row][cow-1]==1 {
                            temp++
                            grid[row][cow-1]=0
                            nextQue = append(nextQue, row*100+cow-1)
                        }
                        if cow!=len(v)-1&&grid[row][cow+1]==1 {
                            temp++
                            grid[row][cow+1]=0
                            nextQue = append(nextQue, row*100+cow+1)
                        }
                    }
                    queue = nextQue
                }
            }
            if temp > max {
                max = temp
            }
        }
    }
    return max
}

 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func maxPathSum(root *TreeNode) int {
    res := math.MinInt32
    helper(root, &res)
    return res
}

func helper(root *TreeNode, res *int) int {
    if root == nil {
        return 0
    }
    left := int(math.Max(float64(helper(root.Left, res)), float64(0)))
    right := int(math.Max(float64(helper(root.Right, res)), float64(0)))
    if root.Val + left + right > *res {
        *res = root.Val + left + right
    }
    return root.Val + int(math.Max(float64(left), float64(right)))
}

func detectCycle(head *ListNode) *ListNode {
    fast, slow:= head, head
    for fast != nil {
        if fast.Next == nil {
            return nil
        }
        slow, fast = slow.Next, fast.Next.Next
        if slow == fast {
            break
        }
    }
    if fast == nil {
        return nil
    }
    for head != fast {
        head, fast = head.Next, fast.Next
    }
    return head
}

func uniquePaths(m int, n int) int {
    store := make([][]int, n)
    for i := range store {
        store[i] = make([]int, m)
    }
    for i:=0; i<n; i++ {
        for j:=0; j<m; j++ {
            if i == j && i == 0 {
                store[i][j] = 1
                continue
            }
            if i != 0 {
                store[i][j] += store[i-1][j]
            }
            if j != 0 {
                store[i][j] += store[i][j-1]
            }
        }
    }
    return store[n-1][m-1]
}

func subsets(nums []int) [][]int {
    queue, res := make([][]int, 0), make([][]int, 1)
    res[0] = []int{}
    for i := range nums {
        queue = append(queue, []int{i})
    }
    for len(queue) > 0 {
        nextQueue := make([][]int, 0)
        for _, v := range queue {
            subset := make([]int, len(v))
            for i, value := range v {
                subset[i] = nums[value]
            }
            res = append(res, subset)
            for j:=v[len(v)-1]+1; j<len(nums); j++ {
                temp := make([]int, len(v))
                copy(temp, v)
                temp = append(temp, j)
                nextQueue = append(nextQueue, temp)
            }
        }
        queue = nextQueue
    }
    return res
}

    var n = uint(len(nums))
    res := make([][]int, 0, 1 << n)
    var i uint
    for i = 0; i < 1<<n; i++ {
        s := make([]int, 0)
        var j uint
        for j = 0; j < n; j++ {
            if i&(1<<j) != 0 {
                s = append(s, nums[j])

        res = append(res, s)
    }

}

func countNodes(root *TreeNode) int {
    if root == nil {
        return 0
    }
    num, success := 0, false
    depth := deep(root, &success, &num)
    return (1<<depth)-1-num
}

func deep(root *TreeNode, success *bool, num *int) int {
    if root.Left == root.Right {
        return 1
    }
    if root.Right == nil {
        (*num)++
        *success = true
        return 2
    }
    right := deep(root.Right, success, num)
    if *success {
        return right+1
    }
    left := deep(root.Left, success, num)
    if left != right {
        (*num) += (1<<right)
        *success = true
    }
    return left+1
}

func change(amount int, coins []int) int {
    dp := make([]int, amount+1)
    dp[0] = 1
    for _, v := range coins {
        for i:=v;i<=amount;i++ {
            dp[i] += dp[i-v] 
        }
    }
    return dp[amount]
}

func surfaceArea(grid [][]int) int {
    res := 0
    min := func (a, b int) int {
        if a < b {
            return a
        }
        return b
    }
    for i, v := range grid {
        for j, value := range v {
            if value != 0 {
                res += (6+(value-1)*4)
                if i != 0 && grid[i-1][j] != 0 {
                    res -= min(value, grid[i-1][j])
                }
                if j != 0 && grid[i][j-1] != 0 {
                    res -= min(value, grid[i][j-1])
                }
                if i != len(grid)-1 && grid[i+1][j] != 0 {
                    res -= min(value, grid[i+1][j])
                }
                if j != len(grid)-1 && grid[i][j+1] != 0 {
                    res -= min(value, grid[i][j+1])
                }
            }
        }
    }
    return res
}

    res := 0
    for i:=0; i<len(grid); i++ {
        for j:=0; j<len(grid); j++ {
            if grid[i][j] > 0 {
                res += 4 * grid[i][j] + 2
            }
            if j>0 {
                res -= min(grid[i][j], grid[i][j-1]) * 2
            }
            if i>0 {
                res -= min(grid[i][j], grid[i-1][j]) * 2
            }
        }
    }

}
