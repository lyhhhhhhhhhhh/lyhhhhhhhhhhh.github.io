func hIndex(citations []int) int {
    m := make(map[int]int)
    for _, value := range citations {
        m[value]++
    }
    store := make([]int, 0)
    for k := range m {
        store = append(store, k)
    }
    sort.Ints(store)
    sum := 0
    for i := len(store) - 1; i >= 0; i-- {
        sum += m[store[i]]
        if sum >= store[i] {
            temp := sum - m[store[i]]
            if i != len(store)-1 && temp >= store[i] {
                return temp
            }
            return store[i]
        }
    }
    if len(citations) > 0 {
        return len(citations)
    }
    return 0
}

func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
    firstLength := getlength(l1)
    secondLength := getlength(l2)
    add := &ListNode{0, nil}
    tail := add
    if firstLength > secondLength {
        for i := 1; i < (firstLength - secondLength); i++ {
            tail.Next = &ListNode{0, nil}
            tail = tail.Next
        }
        tail.Next = l2
        l2 = add
    } else if firstLength < secondLength {
        for i := 1; i < (secondLength - firstLength); i++ {
            tail.Next = &ListNode{0, nil}
            tail = tail.Next
        }
        tail.Next = l1
        l1 = add
    }
    if calculate(l1, l2) == 1 {
        head := &ListNode{1, l1}
        return head
    }
    return l1
}

func getlength(root *ListNode) int {
    res := 0
    for root != nil {
        res++
        root = root.Next
    }
    return res
}

func calculate(l1, l2 *ListNode) int {
    if l1 == nil && l2 == nil {
        return 0
    }
    res := l1.Val + l2.Val + calculate(l1.Next, l2.Next)
    l1.Val = res % 10
    return res / 10
}

func myAtoi(str string) int {
    res, negative, begin := 0, false, false
    for _, value := range str {
        if value == ' ' && !begin {
            continue
        }
        if (value == '+' || value == '-') && !begin {
            if value == '-' {
                negative = true
            }
            begin = true
            continue
        }
        if value >= '0' && value <= '9' {
            if !begin {
                begin = true
            }
            res = res*10 + int(value-48)
            if res > (-math.MinInt32) && negative {
                res = math.MinInt32
                negative = false
                break
            }
            if res > math.MaxInt32 && !negative {
                res = math.MaxInt32
                break
            }
        } else {
            break
        }
    }
    if negative {
        res = (-res)
    }
    return res
}

        }

            count++

            count--
        }

}

这个题属实很巧妙，插入和弹出的O（1）算法都很好写，问题在于最大值上，我们需要额外维持一个数组来保存最大值信息。

最大值的维持有这样一个规律：对于一新的value，之前入队的小于value的值都对其无影响。因此我们维持一个递减数列，每次有新值的时候将递减数列中小于这个值的数全部删除，这样以来我们就可以完成需要的操作。

虽然其中有可能有O（n）的复杂度，但是摊还之后其应当为O（1）。

type MaxQueue struct {
    queue []int
    aux []int
}


func Constructor() MaxQueue {
    return MaxQueue{
        queue: make([]int, 0),
        aux: make([]int, 0),
    }
}


func (this *MaxQueue) Max_value() int {
    if len(this.aux) == 0 {
        return -1
    }
    return this.aux[0]
}


func (this *MaxQueue) Push_back(value int)  {
    this.queue = append(this.queue, value)
    index := 0
    for index = range this.aux {
        if this.aux[index] < value {
            break
        }
    }
    if len(this.aux) > 0 && this.aux[index] >= value { index++ }
    this.aux = append(this.aux[:index], value)
}


func (this *MaxQueue) Pop_front() int {
    if len(this.queue) == 0 {
        return -1
    }
    res := this.queue[0]
    this.queue = this.queue[1:]
    if this.aux[0] == res {
        this.aux = this.aux[1:]
    }
    return res
}

func rotateRight(head *ListNode, k int) *ListNode {
    if head == nil {
        return head
    }
    travel, length := head, 1
    for travel.Next != nil {
        length++
        travel = travel.Next
    }
    travel.Next = head
    for i:=0; i<length-(k%length)-1; i++ {
        head = head.Next
    }
    head, head.Next = head.Next, nil
    return head
}

func maxAreaOfIsland(grid [][]int) int {
    max := 0
    queue:= make([]int, 0)
    for i, v := range grid {
        for j, value := range v {
            temp := 0
            if value == 1 {
                temp++
                grid[i][j] = 0
                queue = append(queue, i*100+j)
                for len(queue) > 0 {
                    nextQue := make([]int, 0)
                    for _, node := range queue {
                        cow, row := node%100, node/100
                        if row!=0 && grid[row-1][cow]==1 {
                            temp++
                            grid[row-1][cow] = 0
                            nextQue = append(nextQue, (row-1)*100+cow)
                        }
                        if row!=len(grid)-1 && grid[row+1][cow]==1 {
                            temp++
                            grid[row+1][cow]=0
                            nextQue = append(nextQue, (row+1)*100+cow)
                        }
                        if cow!=0 && grid[row][cow-1]==1 {
                            temp++
                            grid[row][cow-1]=0
                            nextQue = append(nextQue, row*100+cow-1)
                        }
                        if cow!=len(v)-1&&grid[row][cow+1]==1 {
                            temp++
                            grid[row][cow+1]=0
                            nextQue = append(nextQue, row*100+cow+1)
                        }
                    }
                    queue = nextQue
                }
            }
            if temp > max {
                max = temp
            }
        }
    }
    return max
}

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func maxPathSum(root *TreeNode) int {
    res := math.MinInt32
    helper(root, &res)
    return res
}

func helper(root *TreeNode, res *int) int {
    if root == nil {
        return 0
    }
    left := int(math.Max(float64(helper(root.Left, res)), float64(0)))
    right := int(math.Max(float64(helper(root.Right, res)), float64(0)))
    if root.Val + left + right > *res {
        *res = root.Val + left + right
    }
    return root.Val + int(math.Max(float64(left), float64(right)))
}

func detectCycle(head *ListNode) *ListNode {
    fast, slow:= head, head
    for fast != nil {
        if fast.Next == nil {
            return nil
        }
        slow, fast = slow.Next, fast.Next.Next
        if slow == fast {
            break
        }
    }
    if fast == nil {
        return nil
    }
    for head != fast {
        head, fast = head.Next, fast.Next
    }
    return head
}

这个题是很明显的动态规划问题，递推关系为path[m][n] = path[m-1][n] + path[m][n-1]

func uniquePaths(m int, n int) int {
    store := make([][]int, n)
    for i := range store {
        store[i] = make([]int, m)
    }
    for i:=0; i<n; i++ {
        for j:=0; j<m; j++ {
            if i == j && i == 0 {
                store[i][j] = 1
                continue
            }
            if i != 0 {
                store[i][j] += store[i-1][j]
            }
            if j != 0 {
                store[i][j] += store[i][j-1]
            }
        }
    }
    return store[n-1][m-1]
}

func subsets(nums []int) [][]int {
    queue, res := make([][]int, 0), make([][]int, 1)
    res[0] = []int{}
    for i := range nums {
        queue = append(queue, []int{i})
    }
    for len(queue) > 0 {
        nextQueue := make([][]int, 0)
        for _, v := range queue {
            subset := make([]int, len(v))
            for i, value := range v {
                subset[i] = nums[value]
            }
            res = append(res, subset)
            for j:=v[len(v)-1]+1; j<len(nums); j++ {
                temp := make([]int, len(v))
                copy(temp, v)
                temp = append(temp, j)
                nextQueue = append(nextQueue, temp)
            }
        }
        queue = nextQueue
    }
    return res
}

从0开始每次加一，其二进制是000， 001，010，011，100，101，110，111，刚好和下标对应，是子集的下标。

func subsets(nums []int) [][]int {
    var n = uint(len(nums))
    res := make([][]int, 0, 1 << n)
    var i uint
    for i = 0; i < 1<<n; i++ {
        s := make([]int, 0)
        var j uint
        for j = 0; j < n; j++ {
            if i&(1<<j) != 0 {
                s = append(s, nums[j])
            }
        }
        res = append(res, s)
    }
    return res
}

func buildTree(preorder []int, inorder []int) *TreeNode {
    if len(preorder) == 0 || len(inorder) == 0 {
        return nil
    }

    // 中顺序列找根结点
    var root int
    for k, v :=  range inorder {
        if v == preorder[0] {
            root = k
            break
        }
    }

    // 左右子树归类

    // in_left, in_right := inorder[0: root], inorder[root+1:]

    return &TreeNode{
        Val:   preorder[0],

func buildTree(inorder []int, postorder []int) *TreeNode {
    if len(inorder) == 0 || len(postorder) == 0 {
        return nil
    }
    var root int
    for i, v := range inorder {
        if v == postorder[len(postorder)-1] {
            root = i
            break
        }
    }

    return &TreeNode{
        Val: inorder[root],
        Left: buildTree(inorder[:root], postorder[:root]),
        Right: buildTree(inorder[root+1:], postorder[root:len(postorder)-1]),
    }
}

回溯加剪枝，问题在于如果是一棵满二叉树会有性能退化。

func countNodes(root *TreeNode) int {
    if root == nil {
        return 0
    }
    num, success := 0, false
    depth := deep(root, &success, &num)
    return (1<<depth)-1-num
}

func deep(root *TreeNode, success *bool, num *int) int {
    if root.Left == root.Right {
        return 1
    }
    if root.Right == nil {
        (*num)++
        *success = true
        return 2
    }
    right := deep(root.Right, success, num)
    if *success {
        return right+1
    }
    left := deep(root.Left, success, num)
    if left != right {
        (*num) += (1<<right)
        *success = true
    }
    return left+1
}

func change(amount int, coins []int) int {
    dp := make([]int, amount+1)
    dp[0] = 1
    for _, v := range coins {
        for i:=v;i<=amount;i++ {
            dp[i] += dp[i-v] 
        }
    }
    return dp[amount]
}

        if a < b {
            return a
        }
        return b
    }
    for i, v := range grid {
        for j, value := range v {
            if value != 0 {
                res += (6+(value-1)*4)

                    res -= min(value, grid[i-1][j])
                }
                if j != 0 && grid[i][j-1] != 0 {
                    res -= min(value, grid[i][j-1])
                }
                if i != len(grid)-1 && grid[i+1][j] != 0 {
                    res -= min(value, grid[i+1][j])
                }
                if j != len(grid)-1 && grid[i][j+1] != 0 {
                    res -= min(value, grid[i][j+1])
                }
            }
        }
    }
    return res
}
